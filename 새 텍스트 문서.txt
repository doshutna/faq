    const container = document.getElementById("faqContainer");
    function renderFaqs() {
      container.innerHTML = '';
      const keyword = document.getElementById("searchInput").value.toLowerCase();
      const category = document.getElementById("categorySelect").value;
      faqs.forEach(({ q, a, c }) => {
        const match = q.toLowerCase().includes(keyword);
        const catMatch = !category || c === category;
        if (match && catMatch) {
          const item = document.createElement("div");
          item.className = "faq-item";
          const answerHtml = Array.isArray(a) ? `<ul>${a.map(line => `<li>${line}</li>`).join('')}</ul>` : a;
          item.innerHTML = `
            <div class="question">${q}</div>
            <div class="answer">${answerHtml}</div>
          `;
          item.querySelector('.question').addEventListener('click', function () {
            item.classList.toggle('active');
            item.scrollIntoView({ behavior: 'smooth', block: 'start' });
          });
          container.appendChild(item);
        }
      });
    }

    function addNewFAQ() {
      const q = document.getElementById("newQuestion").value.trim();
      const aRaw = document.getElementById("newAnswer").value.trim();
      const c = document.getElementById("newCategory").value;
      if (!q || !aRaw) return alert("ì§ˆë¬¸ê³¼ ë‹µë³€ì„ ëª¨ë‘ ì…ë ¥í•´ì£¼ì„¸ìš”.");
      const a = aRaw.split(/\n+/).map(line => line.trim()).filter(Boolean);
      faqs.push({ q, a, c });
      saveToLocalStorage();
      renderFaqs();
      document.getElementById("newQuestion").value = "";
      document.getElementById("newAnswer").value = "";
    }

    function saveToLocalStorage() {
      localStorage.setItem("faqData", JSON.stringify(faqs));
    }

    function loadFromLocalStorage() {
      const saved = localStorage.getItem("faqData");
      if (saved) {
        try {
          const parsed = JSON.parse(saved);
          if (Array.isArray(parsed)) {
            faqs.splice(0, faqs.length, ...parsed);
          }
        } catch (e) {
          console.error("FAQ ë¡œë“œ ì˜¤ë¥˜", e);
        }
      }
    }

    document.getElementById("searchInput").addEventListener("input", renderFaqs);
    document.getElementById("categorySelect").addEventListener("change", renderFaqs);
